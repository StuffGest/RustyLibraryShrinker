1//! Gestion de l'extraction des ressources d'images au sein des fichiers EPUB.
2//!
3//! Ce module permet de parcourir la structure d'un livre numérique EPUB pour en
4//! extraire les images tout en préservant scrupuleusement l'ordre alphabétique
5//! et les noms de fichiers originaux.
6//!
7//! Les fonctionnalités incluses :
8//! - Analyse du manifest de l'EPUB.
9//! - Extraction brute des ressources binaires.
10//! - Nettoyage des chemins pour une extraction à plat dans un dossier temporaire.
11
12use anyhow::{Result, anyhow};
13use epub::doc::EpubDoc;
14use std::fs;
15use std::path::Path;
16use std::collections::HashSet;
17
18/// Extrait toutes les images d'un fichier EPUB vers un répertoire cible.
19///
20/// Cette méthode parcourt le manifest des ressources du document pour identifier les
21/// fichiers dont le type MIME correspond à une image, puis les sauvegarde.
22///
23/// # Paramètres
24/// - `epub_path`: Chemin vers le fichier source .epub.
25/// - `temp_dir`: Chemin vers le répertoire où les images seront déposées.
26///
27/// # Erreurs
28/// Retourne une erreur si le fichier est illisible ou si aucune image n'est trouvée.
29pub fn extract_epub(epub_path: &Path, temp_dir: &Path) -> Result<()> {
30    let mut doc = EpubDoc::new(epub_path).map_err(|e| anyhow!("Erreur ouverture EPUB: {:?}", e))?;
31    let mut image_count = 0;
32    let mut seen_resources = HashSet::new();
33
34    // On récupère les ressources d'images directement depuis le manifest.
35    // C'est la méthode la plus fiable pour garder les noms de fichiers originaux
36    // et permettre un tri alphabétique cohérent ultérieurement.
37    let resources = doc.resources.clone();
38
39    for (path, item) in resources {
40        if item.mime.starts_with("image/") {
41            if let Some(data) = doc.get_resource_by_path(&item.path) {
42                save_epub_resource(&path, &data, &item.mime, temp_dir, &mut image_count, &mut seen_resources)?;
43            }
44        }
45    }
46
47    if image_count == 0 {
48        return Err(anyhow!("Aucune image trouvée dans l'EPUB"));
49    }
50
51    Ok(())
52}
53
54/// Sauvegarde une ressource binaire d'image sur le disque.
55///
56/// Nettoie le nom de fichier original pour supprimer les structures de dossiers internes
57/// à l'EPUB (ex: OEBPS/Images/cover.jpg devient cover.jpg) et normalise l'extension.
58///
59/// # Paramètres
60/// - `original_path`: Chemin interne de la ressource dans l'EPUB.
61/// - `data`: Contenu binaire de l'image.
62/// - `mime`: Type MIME de l'image pour validation de l'extension.
63/// - `temp_dir`: Dossier de destination.
64/// - `count`: Compteur mutable d'images traitées.
65/// - `seen`: Set pour garantir l'unicité des fichiers extraits.
66fn save_epub_resource(
67    original_path: &str,
68    data: &[u8],
69    mime: &str,
70    temp_dir: &Path,
71    count: &mut usize,
72    seen: &mut HashSet<String>
73) -> Result<()> {
74    // On extrait uniquement le nom du fichier pour une extraction à plat
75    let file_name = original_path.split('/').last().unwrap_or(original_path).to_string();
76
77    // Protection contre les doublons de ressources
78    if !seen.insert(file_name.clone()) {
79        return Ok(());
80    }
81
82    *count += 1;
83
84    // Détermination de l'extension correcte basée sur le type MIME
85    let ext = match mime {
86        "image/jpeg" | "image/jpg" => "jpg",
87        "image/png" => "png",
88        "image/webp" => "webp",
89        _ => file_name.split('.').last().unwrap_or("jpg"),
90    };
91
92    // On garde le NOM ORIGINAL pour respecter le tri alphabétique du comic.
93    // On s'assure juste que l'extension est cohérente avec la donnée.
94    let base_name = if let Some(pos) = file_name.rfind('.') {
95        &file_name[..pos]
96    } else {
97        &file_name
98    };
99
100    let dest_name = format!("{}.{}", base_name, ext);
101    fs::write(temp_dir.join(dest_name), data)?;
102    Ok(())
103}
