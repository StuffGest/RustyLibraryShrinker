1//! Gestion de l'extraction des archives de bandes dessinées.
2//!
3//! Ce module centralise les mécanismes d'extraction pour les différents formats
4//! de fichiers supportés par l'application :
5//! - Archives compressées (CBZ, CBR) via `zip_rar`
6//! - Documents PDF via `pdf`
7//! - Livres numériques EPUB via `epub`
8//!
9//! Il fournit une interface unifiée pour décompresser ces fichiers dans des
10//! répertoires temporaires avant le traitement des images.
11
12pub mod zip_rar;
13pub mod pdf;
14pub mod epub;
15
16use anyhow::Result;
17use std::path::Path;
18use crate::models::{ComicFile, ComicType};
19use indicatif::ProgressBar;
20
21/// Point d'entrée unique pour l'extraction de fichiers de bande dessinée.
22///
23/// Cette fonction identifie le format du fichier via son `ComicType` et délègue
24/// l'extraction au sous-module approprié.
25///
26/// # Paramètres
27/// - `comic`: Une référence vers la structure `ComicFile` contenant le chemin et le type du fichier.
28/// - `temp_dir`: Le chemin du répertoire temporaire où les images doivent être extraites.
29/// - `_progress`: Une instance de `ProgressBar` pour le suivi visuel de l'opération.
30///
31/// # Erreurs
32/// Retourne une erreur si le format est corrompu ou si l'accès au système de fichiers échoue.
33pub fn extract_comic(comic: &ComicFile, temp_dir: &Path, _progress: &ProgressBar) -> Result<()> {
34    match comic.file_type {
35        ComicType::Cbz => zip_rar::extract_zip(&comic.path, temp_dir),
36        ComicType::Cbr => {
37            // Tente l'extraction RAR. Si elle échoue, tente l'extraction ZIP car
38            // de nombreux fichiers .cbr sont en réalité des archives ZIP renommées.
39            if zip_rar::extract_rar(&comic.path, temp_dir).is_err() {
40                zip_rar::extract_zip(&comic.path, temp_dir)
41            } else {
42                Ok(())
43            }
44        },
45        ComicType::Pdf => pdf::extract_pdf(&comic.path, temp_dir),
46        ComicType::Epub => epub::extract_epub(&comic.path, temp_dir),
47    }
48}
