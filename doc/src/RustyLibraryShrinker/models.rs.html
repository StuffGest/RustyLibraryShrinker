21/// Traduit une clé Fluent avec des arguments optionnels et une langue dynamique
22pub fn tr(key: &str, lang: &str, args: Option<&HashMap<String, FluentValue>>) -> String {
23    let lang_id: fluent_templates::LanguageIdentifier = lang.parse().unwrap_or_else(|_| "en".parse().unwrap());
24
25    match args {
26        Some(fluent_args) => {
27            LOCALES.lookup_with_args(&lang_id, key, fluent_args)
28        }
29        None => {
30            LOCALES.lookup(&lang_id, key)
31        }
32    }
33}

35/// Détecte dynamiquement la langue du système ou retombe sur le français par défaut.
36pub fn detect_system_lang() -> String {
37    std::env::var("LANG")
38        .or_else(|_| std::env::var("LC_ALL"))
39        .unwrap_or_else(|_| "fr".to_string())
40        .split('.')
41        .next()
42        .unwrap_or("fr")
43        .to_string()
44}

62/// Modes de gestion des fichiers de sortie après traitement.
63#[derive(Debug, Clone, Copy, ValueEnum, PartialEq)]
64pub enum FileMode {
65    /// Crée un nouveau fichier avec un suffixe ".optimise.cbz" (Défaut).
66    Suffix,
67    /// Renomme l'original en ".original.cbz" et donne le nom d'origine au compressé.
68    Rename,
69    /// Remplace directement l'original (pas de sauvegarde/backup).
70    Replace,
71}

95/// Configuration globale de l'application extraite des arguments CLI.
96///
97/// Gère les paramètres de qualité d'image, les dimensions de sortie
98/// et les options de manipulation de fichiers.
99#[derive(Parser, Debug, Clone)]
100#[command(author, version, about = help_about(), long_about = None)]
101pub struct Args {
102    /// Chemin vers le fichier unique ou le répertoire contenant les archives à traiter.
103    #[arg(value_name = "INPUT", help = help_input())]
104    pub input: Option<PathBuf>,
105
106    /// Niveau de qualité pour l'encodage WebP (de 1 à 100). Une valeur de 90 offre un excellent rapport poids/qualité.
107    #[arg(long, default_value = "fr", help = help_lang())]
108    pub lang: String,
109
110    /// Niveau de qualité pour l'encodage WebP (de 1 à 100). Une valeur de 90 offre un excellent rapport poids/qualité.
111    #[arg(short, long, default_value = "90", help = help_quality())]
112    pub quality: u8,
113
114    /// Hauteur cible en pixels pour le redimensionnement des pages.
115    #[arg(short = 'H', long, default_value = "1800", help = help_height())]
116    pub target_height: u32,
117
118    /// Dimension maximale de sécurité utilisée en cas d'impossibilité de déterminer le ratio.
119    #[arg(short, long, default_value = "1200", help = help_dim())]
120    pub max_dimension: u32,
121
122    /// Mode de gestion des fichiers de sortie (suffix, rename, replace).
123    #[arg(short = 'r', long, value_enum, default_value_t = FileMode::Suffix, help = help_mode())]
124    pub file_mode: FileMode,
125
126    /// Nombre de threads maximum (0 pour auto)
127    #[arg(short, long, default_value_t = 0, help = help_threads())]
128    pub threads: usize,
129
130    /// Utilisation d'un motif de recherche Glob pour filtrer les fichiers (ex: "**/Batman*.cbr").
131    #[arg(short, long, help = help_glob())]
132    pub glob_pattern: Option<String>,
133
134    /// Pourcentage minimal de gain de poids requis pour valider le remplacement du fichier original.
135    #[arg(long, default_value = "5.0", help = help_savings())]
136    pub min_savings: f64,
137
138    /// Chemin vers le fichier de log
139    #[arg(short = 'l', long = "log-file", help = help_log())]
140    pub log_file: Option<PathBuf>,
141
142    /// Affiche plus d'informations dans la console durant l'exécution.
143    #[arg(short, long, help = help_verbose())]
144    pub verbose: bool,
145
146    /// Désactive la compression des images : convertit uniquement le conteneur vers le format CBZ.
147    #[arg(short = 'S', long, help = help_skip())]
148    pub skip_compression: bool,
149
150    /// Force le ré-encodage et le redimensionnement des images même si elles sont déjà au format WebP.
151    #[arg(long, help = help_force())]
152    pub force_shrink: bool,
153}

155/// Résultat détaillé du traitement d'un fichier.
156///
157/// Utilisé pour générer le rapport final à l'utilisateur.
158#[allow(dead_code)]
159#[derive(Debug)]
160pub struct ProcessingStats {
161    /// Taille du fichier original en octets.
162    pub original_size: u64,
163    /// Taille du fichier généré après optimisation.
164    pub compressed_size: u64,
165    /// Nombre d'images traitées avec succès.
166    pub images_processed: usize,
167    /// Nombre d'images ignorées (erreurs ou formats non supportés).
168    pub images_skipped: usize,
169    /// Indique si la compression a été annulée faute de gain suffisant.
170    pub compression_skipped: bool,
171    /// Chemin final du fichier généré.
172    pub output_path: Option<PathBuf>,
173    /// Message détaillé en cas d'erreur lors du pipeline.
174    pub error_message: Option<String>,
175    /// Message de statut informatif (ex: "Gain insuffisant").
176    pub status_message: Option<String>,
177    /// Nom image ignoree
178    pub skipped_details: Vec<(String, String)>,
179}
