1//! Définitions des structures de données et des types fondamentaux.
2//!
3//! Ce module contient les types partagés à travers toute l'application :
4//! - Les énumérations des formats supportés.
5//! - La structure de configuration issue de la ligne de commande.
6//! - Les modèles de suivi des statistiques de traitement.
7
8use clap::{Parser, ValueEnum};
9use std::path::PathBuf;
10
11/// Modes de gestion des fichiers de sortie après traitement.
12#[derive(Debug, Clone, Copy, ValueEnum, PartialEq)]
13pub enum FileMode {
14    /// Crée un nouveau fichier avec un suffixe ".optimise.cbz" (Défaut).
15    Suffix,
16    /// Renomme l'original en ".original.cbz" et donne le nom d'origine au compressé.
17    Rename,
18    /// Remplace directement l'original (pas de sauvegarde/backup).
19    Replace,
20}
21
22/// Formats de fichiers de bande dessinée pris en charge par l'outil.
23#[derive(Debug, Clone)]
24pub enum ComicType {
25    /// Comic Book Zip : Archives au format ZIP.
26    Cbz,
27    /// Comic Book Rar : Archives au format RAR.
28    Cbr,
29    /// Portable Document Format.
30    Pdf,
31    /// Electronic Publication : Livres numériques.
32    Epub,
33}
34
35/// Représente un fichier de bande dessinée identifié pour le traitement.
36#[derive(Debug, Clone)]
37pub struct ComicFile {
38    /// Chemin absolu ou relatif vers le fichier sur le disque.
39    pub path: PathBuf,
40    /// Type de fichier détecté par son extension.
41    pub file_type: ComicType,
42}
43
44/// Configuration globale de l'application extraite des arguments CLI.
45///
46/// Gère les paramètres de qualité d'image, les dimensions de sortie
47/// et les options de manipulation de fichiers.
48#[derive(Parser, Debug, Clone)]
49#[command(author, version, about = "Compresseur de BD, Mangas et Comics vers CBZ", long_about = None)]
50pub struct Args {
51    /// Chemin vers le fichier unique ou le répertoire contenant les archives à traiter.
52    #[arg(value_name = "INPUT")]
53    pub input: Option<PathBuf>,
54
55    /// Niveau de qualité pour l'encodage WebP (de 1 à 100). Une valeur de 90 offre un excellent rapport poids/qualité.
56    #[arg(short, long, default_value = "90")]
57    pub quality: u8,
58
59    /// Hauteur cible en pixels pour le redimensionnement des pages.
60    #[arg(short = 'H', long, default_value = "1800")]
61    pub target_height: u32,
62
63    /// Dimension maximale de sécurité utilisée en cas d'impossibilité de déterminer le ratio.
64    #[arg(short, long, default_value = "1200")]
65    pub max_dimension: u32,
66
67    /// Mode de gestion des fichiers de sortie (suffix, rename, replace).
68    #[arg(short = 'r', long, value_enum, default_value_t = FileMode::Suffix)]
69    pub file_mode: FileMode,
70
71    /// Nombre de threads maximum (0 pour auto)
72    #[arg(short, long, default_value_t = 0)]
73    pub threads: usize,
74
75    /// Utilisation d'un motif de recherche Glob pour filtrer les fichiers (ex: "**/Batman*.cbr").
76    #[arg(short, long)]
77    pub glob_pattern: Option<String>,
78
79    /// Pourcentage minimal de gain de poids requis pour valider le remplacement du fichier original.
80    #[arg(long, default_value = "5.0")]
81    pub min_savings: f64,
82
83    /// Chemin vers le fichier de log
84    #[arg(short = 'l', long = "log-file")]
85    pub log_file: Option<PathBuf>,
86
87    /// Affiche plus d'informations dans la console durant l'exécution.
88    #[arg(short, long)]
89    pub verbose: bool,
90
91    /// Désactive la compression des images : convertit uniquement le conteneur vers le format CBZ.
92    #[arg(short = 'S', long)]
93    pub skip_compression: bool,
94}
95
96/// Résultat détaillé du traitement d'un fichier.
97///
98/// Utilisé pour générer le rapport final à l'utilisateur.
99#[allow(dead_code)]
100#[derive(Debug)]
101pub struct ProcessingStats {
102    /// Taille du fichier original en octets.
103    pub original_size: u64,
104    /// Taille du fichier généré après optimisation.
105    pub compressed_size: u64,
106    /// Nombre d'images traitées avec succès.
107    pub images_processed: usize,
108    /// Nombre d'images ignorées (erreurs ou formats non supportés).
109    pub images_skipped: usize,
110    /// Indique si la compression a été annulée faute de gain suffisant.
111    pub compression_skipped: bool,
112    /// Chemin final du fichier généré.
113    pub output_path: Option<PathBuf>,
114    /// Message détaillé en cas d'erreur lors du pipeline.
115    pub error_message: Option<String>,
116    /// Message de statut informatif (ex: "Gain insuffisant").
117    pub status_message: Option<String>,
118    /// Nom image ignoree
119    pub skipped_details: Vec<(String, String)>,
120}
