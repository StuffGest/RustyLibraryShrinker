20        locales: "./locales",
21        fallback_language: "en",

25/// Traduit une clé Fluent avec des arguments optionnels et une langue dynamique
26pub fn tr(key: &str, lang: &str, args: Option<&HashMap<String, FluentValue>>) -> String {
27    // Vérification de sécurité : si la langue n'est pas supportée, on force le fallback
28    let safe_lang = if SUPPORTED_LANGS.contains(&lang) { lang } else { DEFAULT_LANG };
29    let lang_id: fluent_templates::LanguageIdentifier = safe_lang.parse().unwrap_or_else(|_| DEFAULT_LANG.parse().unwrap());
30
31    match args {
32        Some(fluent_args) => {
33            LOCALES.lookup_with_args(&lang_id, key, fluent_args)
34        }
35        None => {
36            LOCALES.lookup(&lang_id, key)
37        }
38    }
39}

41/// Détecte dynamiquement la langue du système ou retombe sur l'anglais par défaut.
42pub fn detect_system_lang() -> String {
43    let lang_var = std::env::var("LANG")
44        .or_else(|_| std::env::var("LC_ALL"))
45        .unwrap_or_else(|_| DEFAULT_LANG.to_string());
46
47    lang_var.split('.')
48        .next()
49        .map(|s| s.split('_').next().unwrap_or(DEFAULT_LANG))
50        .unwrap_or(DEFAULT_LANG)
51        .to_string()
52}

71/// Modes de gestion des fichiers de sortie après traitement.
72#[derive(Debug, Clone, Copy, ValueEnum, PartialEq)]
73pub enum FileMode {
74    /// Crée un nouveau fichier avec un suffixe ".optimise.cbz" (Défaut).
75    Suffix,
76    /// Renomme l'original en ".original.cbz" et donne le nom d'origine au compressé.
77    Rename,
78    /// Remplace directement l'original (pas de sauvegarde/backup).
79    Replace,
80}

104/// Configuration globale de l'application extraite des arguments CLI.
105#[derive(Parser, Debug, Clone)]
106#[command(author, version, about = help_about(DEFAULT_LANG))]
107pub struct Args {
108    /// Chemin vers le fichier unique ou le répertoire contenant les archives à traiter.
109    #[arg(value_name = "INPUT", help = help_input(DEFAULT_LANG))]

111
112    /// Langue de l'interface (fr, en).

114    pub lang: String,
115
116    /// Niveau de qualité pour l'encodage WebP (de 1 à 100).
117    #[arg(short, long, default_value = "90", help = help_quality(DEFAULT_LANG))]
118    pub quality: u8,
119
120    /// Hauteur cible en pixels pour le redimensionnement des pages.
121    #[arg(short = 'H', long, default_value = "1800", help = help_height(DEFAULT_LANG))]
122    pub target_height: u32,
123
124    /// Dimension maximale de sécurité utilisée en cas d'impossibilité de déterminer le ratio.
125    #[arg(short, long, default_value = "1200", help = help_dim(DEFAULT_LANG))]
126    pub max_dimension: u32,
127
128    /// Mode de gestion des fichiers de sortie (suffix, rename, replace).
129    #[arg(short = 'r', long, value_enum, default_value_t = FileMode::Suffix, help = help_mode(DEFAULT_LANG))]

131
132    /// Nombre de threads maximum (0 pour auto)
133    #[arg(short, long, default_value_t = 0, help = help_threads(DEFAULT_LANG))]
134    pub threads: usize,
135
136    /// Utilisation d'un motif de recherche Glob pour filtrer les fichiers.
137    #[arg(short, long, help = help_glob(DEFAULT_LANG))]
138    pub glob_pattern: Option<String>,
139
140    /// Pourcentage minimal de gain de poids requis.
141    #[arg(long, default_value = "5.0", help = help_savings(DEFAULT_LANG))]
142    pub min_savings: f64,
143
144    /// Chemin vers le fichier de log
145    #[arg(short = 'l', long = "log-file", help = help_log(DEFAULT_LANG))]
146    pub log_file: Option<PathBuf>,
147
148    /// Affiche plus d'informations dans la console durant l'exécution.
149    #[arg(short, long, help = help_verbose(DEFAULT_LANG))]
150    pub verbose: bool,
151
152    /// Désactive la compression des images.
153    #[arg(short = 'S', long, help = help_skip(DEFAULT_LANG))]
154    pub skip_compression: bool,
155
156    /// Force le ré-encodage des images même si elles sont déjà en WebP.
157    #[arg(long, help = help_force(DEFAULT_LANG))]

159}
